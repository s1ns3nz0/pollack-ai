## S1. GPS/GNSS 스푸핑
### ① 시나리오 설명
**자산**: GNSS (Tier 1) · **심각도**: H
정찰 → EW로 위치 선점 → 위조 GNSS 신호 주입 → 위치 drift 유발 → 항로 이탈.
GPS와 INS(관성항법)는 서로 독립적인 위치 추정 수단이다. ArduPilot은 이 둘을 EKF(확장칼만필터)로 융합하는데, 스푸핑이 들어가면 GPS가 말하는 위치/속도와 INS 추정치가 어긋나 **EKF 잔차(variance)가 급등**한다. 따라서 본 룰은 스푸핑 신호 자체가 아니라 그 부작용인 EKF 잔차 이상을 탐지한다.
- 핵심 신호: `PosHorizVariance`(수평 위치 잔차), `VelocityVariance`(속도 잔차)
- 보조 신호: `FixType`(GPS 측위 품질), `SatellitesVisible`(위성 수), `Eph_cm`(GPS 수평 오차)
**MITRE**: ICS T0830·T0856·T0831·T0815 / EMB3D 센서변조 (Enterprise 매핑: T1557 Collection, T1565·T1499 Impact)
### ② Watchlist 설명 — `GNSS_Exception_List`

### ③ KQL 설명
고정 임계값 대신 **동적 게이트**를 사용한다. 최근 24시간 정상 데이터의 P99에 1.5배를 곱해 임계를 자동 산출하며, 환경 변화에 자동 추종한다. 게이트 계산 시 `잔차 < 1.0`만 사용해 공격값이 baseline에 섞여 임계가 상승하는 오염을 방지한다. 통계 베이스라인은 24시간, 실제 알림 판단은 최근 10분으로 분리한다.
```plain text
let ExceptionList = _GetWatchlist('GNSS_Exception_List');
let baselineWindow = 24h;        // 패턴 학습 범위
let lookbackWindow = 5m;         // 5분 단위로 묶어서 봄
let alertWindow = 10m;           // 알림은 최근 10분 것만
let zScoreThreshold = 3.0;       // 평균에서 3σ 벗어나면 이상치
let gateMultiplier = 1.5;        // 게이트 = P99 x 1.5
let attackFloor = 1.0;           // 1.0 이상은 공격으로 보고 baseline 제외

// 동적 게이트: 정상 데이터(공격값 제외) P99 x 1.5 자동 산출
let dynamicPosGate = toscalar(
    UAVTelemetry_CL
    | where TimeGenerated > ago(baselineWindow)
    | where MsgType == "EKF_STATUS_REPORT"
    | where PosHorizVariance < attackFloor
    | summarize gate = percentile(PosHorizVariance, 99) * gateMultiplier);
let dynamicVelGate = toscalar(
    UAVTelemetry_CL
    | where TimeGenerated > ago(baselineWindow)
    | where MsgType == "EKF_STATUS_REPORT"
    | where VelocityVariance < attackFloor
    | summarize gate = percentile(VelocityVariance, 99) * gateMultiplier);
// 게이트 하한 보호: baseline이 너무 조용해도 최소 0.02 유지
let posVarGate = max_of(dynamicPosGate, 0.02);
let velVarGate = max_of(dynamicVelGate, 0.02);

// 보조 신호: 같은 윈도우 GPS 품질 집계
let GpsQuality = UAVTelemetry_CL
    | where TimeGenerated > ago(baselineWindow)
    | summarize MinFixType = min(FixType), MinSats = min(SatellitesVisible), MaxEph = max(Eph_cm)
      by UAVId, TimeBin = bin(TimeGenerated, lookbackWindow);

// 주 신호: EKF 잔차 이상 (z-score 자동 추종)
UAVTelemetry_CL
| where TimeGenerated > ago(baselineWindow)
| where MsgType == "EKF_STATUS_REPORT"
| make-series AvgPosVar = avg(PosHorizVariance), AvgVelVar = avg(VelocityVariance)
    on TimeGenerated step lookbackWindow by UAVId
| extend (PosAnomalies, PosScore, PosBaseline) = series_decompose_anomalies(AvgPosVar, zScoreThreshold)
| extend (VelAnomalies, VelScore, VelBaseline) = series_decompose_anomalies(AvgVelVar, zScoreThreshold)
| mv-expand TimeGenerated, AvgPosVar, AvgVelVar, PosAnomalies, PosScore, VelAnomalies, VelScore
| where TimeGenerated > ago(alertWindow)
| where PosAnomalies == 1 and VelAnomalies == 1                       // 위치+속도 둘 다 이상
| where todouble(AvgPosVar) > posVarGate and todouble(AvgVelVar) > velVarGate  // 동적 게이트 초과
// watchlist 예외: Normal_Operations가 아닌 의도적 예외상황만 억제
| join kind=leftouter (
    ExceptionList
    | where Condition_s != "Normal_Operations"
    | project UAVId = UAVId_s, MaxPosResidual_d, MaxVelResidual_d, ExceptionCondition = Condition_s
) on UAVId
| extend isPosException = isnotnull(MaxPosResidual_d) and todouble(AvgPosVar) < todouble(MaxPosResidual_d)
| extend isVelException = isnotnull(MaxVelResidual_d) and todouble(AvgVelVar) < todouble(MaxVelResidual_d)
| where not(isPosException and isVelException)
// 보조 신호 join → 신뢰도 가산점
| extend TimeBin = bin(todatetime(TimeGenerated), lookbackWindow)
| join kind=leftouter (GpsQuality) on UAVId, TimeBin
| extend CorroborationScore =
      toint(MinFixType < 3)              // GPS fix 3D 미만
    + toint(MinSats < 6)                 // 위성 수 비정상 하락
    + toint(todouble(MaxEph) > 500)      // GPS 수평오차 5m 초과
| extend ConfidenceLevel = case(CorroborationScore >= 2, "High", CorroborationScore == 1, "Medium", "Low")
| extend TriggerReason = "Both_PosAndVel_ZScore_AboveDynamicGate"
| extend DetectionDetail = strcat(
    "PosVar:", todouble(AvgPosVar), "|PosGate:", posVarGate, "|PosScore:", todouble(PosScore),
    "|VelVar:", todouble(AvgVelVar), "|VelGate:", velVarGate,
    "|FixType:", MinFixType, "|Sats:", MinSats, "|Eph_cm:", MaxEph,
    "|ExceptionCond:", coalesce(ExceptionCondition, "none"))
| project TimeGenerated = todatetime(TimeGenerated), UAVId, TriggerReason, ConfidenceLevel,
    CorroborationScore, DetectionDetail, AvgPosVar = todouble(AvgPosVar), AvgVelVar = todouble(AvgVelVar),
    PosGate = posVarGate, VelGate = velVarGate, PosScore = todouble(PosScore)
| sort by TimeGenerated desc
```

### ⑤ 오탐/정탐
기준값 (정상 비행 25만 샘플 실측): PosHorizVariance P99 = 0.0159, VelocityVariance P99 = 0.0104 → 동적 게이트 자동 산출값 PosGate 0.0238 / VelGate 0.0156
- **오탐**: 이착륙·급기동 순간 PosHorizVariance가 일시적으로 0.02\~0.05까지 튐. z-score는 잡힐 수 있으나 보조신호(FixType/Sats/Eph)가 정상이라 `CorroborationScore=0`, `ConfidenceLevel=Low` → Triage Agent가 저신뢰로 분류, 자동 종결 가능
- **정탐**: PosHorizVariance 0.5↑ + VelocityVariance 동반 급등 + FixType 3D 미만 하락 + 위성 수 급감 → z-score AND 동적 게이트 동시 충족 + `CorroborationScore=2~3`, `ConfidenceLevel=High` → `TriggerReason=Both_PosAndVel_ZScore_AboveDynamicGate` → 즉시 정탐, RTB 플레이북 실행
> 검증 과제: MaxPosVar 0.259(정상 중 일시 스파이크)의 FP 여부는 실제 공격 데이터로 TP/FP 확인 필요. 필요 시 N분 지속조건 추가 검토.

## S2. C2 재밍·하이재킹
### ① 시나리오 설명
**자산**: C2_LINK (Tier 1) · **심각도**: H<br>
**C2 링크** = 조종기랑 드론 사이의 "리모컨 신호선". 이게 뚫리면 드론이 누구 명령이든 다 들어버려.
**공격 흐름**: 재밍(리모컨 신호 끊기) → 세션 하이재킹(끊긴 틈에 자기가 조종기인 척 접속) → 명령주입(가짜 명령 보내기) → 통제권 탈취.
<br>**MITRE**: ICS T0814·T0855·T0831·T0813 / EMB3D 약한인증

### ③ KQL 설명
시퀀스 역행(Replay)은 "환경과 무관하게 즉시 정탐"이어야 하므로 평균값으로 뭉개지는 통계 블록이 아니라 원본 로그를 그대로 보는 즉시탐지 블록에 둠. 점프(증가폭 이상치)만 통계 블록에서 베이스라인 24h로 탐지.
```plain text
// C2 하이재킹 탐지: "즉시 확인해야 할 것"과 "패턴으로 봐야 할 것"을 분리해서 따로 처리
// ★ 재밍지표에 지형/환경 예외 Watchlist 추가 (S1과 같은 패턴 — 신원/리플레이는 예외 적용 안 함)

let whitelistedGcsIds = _GetWatchlist('C2_Whitelisted_GCS_List');         // UAV별 허용된 지상국(GCS) 짝꿍 목록
let RFExceptionList = _GetWatchlist('C2_RF_Environment_Exception_List');  // 지역별 "원래 신호 약한 구간" 허용치
let lookback = 1h;
let binSize = 5m;

// [블록 A] 즉시 탐지: 비인가 GCS / 재밍 / 시퀀스 역행 — 평균 내지 않고 원본 그대로 봄
let InstantDetection =
    UAVTelemetry_CL
    | where TimeGenerated > ago(lookback)
    | sort by UAVId_s, TimeGenerated asc
    | serialize
    | extend PrevUAV = prev(UAVId_s, 1), PrevSeq = prev(MavlinkSeqNum_d, 1)
    | extend SeqDelta = iff(UAVId_s == PrevUAV, MavlinkSeqNum_d - PrevSeq, real(null))
    | join kind=leftouter (whitelistedGcsIds) on UAVId_s
    | extend isUnauth = iff(MavlinkSourceSystemId_d != SystemId_d or isnull(SystemId_d), true, false)
        // ★ 신원 기반이라 예외 적용 안 함 — 무조건 즉시 알람
    | join kind=leftouter (RFExceptionList) on Region_s
        // ★ 추가: 이 지역, "원래 신호 약한 구간" 목록에 있나 대조
    | extend isRFException = isnotnull(MinRssiDbm_d)
        and C2LinkRssiDbm_d >= MinRssiDbm_d
        and C2LinkPacketLossPct_d <= MaxPacketLossPct_d
        // ★ 등록범위 "안"일 때만 예외. 넘으면 예외 아님
    | extend isJamming = iff(not(isRFException) and C2LinkRssiDbm_d <= -20.0 and C2LinkPacketLossPct_d >= 30.0, true, false)
    | extend isReplay = iff(SeqDelta < 0, true, false)
        // ★ 이것도 예외 적용 안 함 — 시퀀스 역행은 환경과 무관
    | where isUnauth or isJamming or isReplay
    | project TimeGenerated, UAVId_s,
        DetectionType = case(isUnauth, "Unauthorized_Source", isJamming, "Jamming_Indicator", "Sequence_Replay"),
        Detail = strcat("Unauth:", isUnauth, "|Jamming:", isJamming, "|Replay:", isReplay);

// [블록 B] 통계 탐지: 시퀀스 점프 (베이스라인 24h로 학습, 알림은 최근 10분만)
let baselineWindow = 24h;
let alertWindow = 10m;
let SeqDetection =
    UAVTelemetry_CL
    | where TimeGenerated > ago(baselineWindow)
    | sort by UAVId_s, TimeGenerated asc
    | serialize
    | extend PrevUAV = prev(UAVId_s, 1), PrevSeq = prev(MavlinkSeqNum_d, 1)
    | extend SeqDelta = iff(UAVId_s == PrevUAV, MavlinkSeqNum_d - PrevSeq, real(null))
    | make-series AvgSeqDelta = avg(SeqDelta) on TimeGenerated step binSize by UAVId_s
    | extend (SeqDeltaAnomalies, Score, Baseline) = series_decompose_anomalies(AvgSeqDelta, 3.0)
    | mv-expand TimeGenerated, AvgSeqDelta, SeqDeltaAnomalies
    | where TimeGenerated > ago(alertWindow) and SeqDeltaAnomalies == 1 and AvgSeqDelta > 0
    | project TimeGenerated, UAVId_s, DetectionType = "Sequence_Jump", Detail = strcat("Delta:", AvgSeqDelta);

union InstantDetection, SeqDetection
| sort by TimeGenerated desc
```

## S3. SATCOM MITM
### ① 시나리오 설명

### ③ KQL 설명
다른 시나리오와 같은 패턴 보정만 적용 (새 버그는 없었음): 베이스라인/알림창 분리, `coalesce` 대신 `isnotnull()` 명시.
```plain text
// SATCOM MITM 탐지: "지연시간(환경 탓일 수 있음)"과 "체크섬 불일치(변조, 환경과 무관)"를 구분
// MAC 인증실패는 "지연도 같이 튀고 등록된 구간"일 때만 날씨로 설명 가능, 그 외엔 체크섬처럼 무조건 정탐

let ExceptionList = _GetWatchlist('SATCOM_Exception_List');  // 터미널별 "이정도 지연은 정상" 허용치
let baselineWindow = 24h;
let alertWindow = 10m;
let binSize = 5m;

SATCOMLink_CL
| where TimeGenerated > ago(baselineWindow)
| extend IsMacFailure = iff(MacVerificationResult_s == "fail", 1, 0)
    // 인증코드 검증 실패를 1/0 숫자로 바꿔서 나중에 평균(=실패율)을 낼 수 있게 함
| make-series
    FailRate = avg(IsMacFailure),                                   // 5분당 인증실패율 흐름
    ChecksumFailCount = countif(PayloadChecksumMatch_b == false),    // 5분당 체크섬 불일치 횟수
    AvgLatency = avg(Latency_d)                                      // 5분당 평균 지연시간
    on TimeGenerated step binSize by UAVId_s, SatcomTerminalId_s
| extend (FailAnomalies, Score, Baseline) = series_decompose_anomalies(FailRate, 3.0)
    // 인증실패율이 평소보다 튀는 지점 찾기
| mv-expand TimeGenerated, FailRate, ChecksumFailCount, AvgLatency, FailAnomalies
| where TimeGenerated > ago(alertWindow)
    // 알림은 최근 10분 것만

| join kind=leftouter (ExceptionList) on UAVId_s, SatcomTerminalId_s
    // 이 터미널, 허용 지연시간 목록에 있나 대조
| extend isWeatherExplainable = isnotnull(MaxLatency_d) and AvgLatency > MaxLatency_d and ChecksumFailCount == 0
    // 등록된 지역 + 지연도 같이 튀고 + 체크섬은 안 깨짐 = "신호감쇄로 설명 가능한" 정황
| extend isSuppressedMacFailure = FailAnomalies == 1 and isWeatherExplainable
    // MAC 인증실패가 그 정황 안에 있을 때만 억제 (지연이 정상이면 정황 자체가 없어서 억제 안 됨)
| where (FailAnomalies == 1 and not(isSuppressedMacFailure)) or ChecksumFailCount > 0
    // 체크섬 불일치는 무조건 통과. MAC 실패는 "날씨로 설명 안 될 때만" 통과
| project TimeGenerated, UAVId_s, SatcomTerminalId_s, AvgLatency, ChecksumFailCount, FailAnomalies
```

### ③ KQL 설명
Watchlist를 join만 하고 실제 분기에 안 쓰던 부분을 고쳐서, `KnownGoodHash_s`와 실제 해시를 비교해 "공식 채널의 알려진 정상 해시"인 경우만 예외로 인정하도록 함.
```plain text
// 펌웨어 무결성 탐지: 서명·해시·SBOM 중 하나라도 어긋나면 위협으로 간주 (Zero-Tolerance)

let IntegrityWhitelist = _GetWatchlist('Firmware_Integrity_Whitelist');  // 모델+채널별 "알려진 정상 해시" 목록
let scanWindow = 5m;

UAVFirmwareIntegrity_CL
| where TimeGenerated > ago(scanWindow)
| join kind=leftouter (IntegrityWhitelist) on DeviceModel_s, UpdateChannel_s
    // 이 모델+채널 조합의 "정상 해시"가 뭔지 대조
| extend isKnownGoodHash = isnotnull(KnownGoodHash_s) and FirmwareHash_s == KnownGoodHash_s
    // isnotnull() 명시: 와치리스트에 안 등록된 모델이면 KnownGoodHash_s가 null이라
    // "FirmwareHash_s == null"이 true/false가 아니라 null로 새는 것 방지
| extend isIntegrityViolation = iff(
    FirmwareSignatureValid_b == false                              // 서명 자체가 무효하면 → 위반
    or SbomUnregisteredComponentFound_b == true                    // 등록 안 된 컴포넌트 발견되면 → 위반
    or (FirmwareHashMatch_b == false and not(isKnownGoodHash)),     // 해시 불일치인데 알려진 정상해시도 아니면 → 위반
    true, false)
| where isIntegrityViolation
    // 위 셋 중 하나라도 해당하면 통과 (베타채널의 알려진 정상해시는 여기서 자동으로 빠짐)
| extend ViolationCount = toint(FirmwareSignatureValid_b == false)
    + toint(SbomUnregisteredComponentFound_b == true)
    + toint(FirmwareHashMatch_b == false and not(isKnownGoodHash))
    // 위반이 몇 개나 동시에 일어났는지 세기 (정보손실 방지)
| extend DetectionType = case(
    ViolationCount > 1, "Multiple_Violations",
    FirmwareSignatureValid_b == false, "Signature_Mismatch",
    SbomUnregisteredComponentFound_b == true, "SBOM_Unregistered_Component",
    "Hash_Mismatch_Unverified"
)
| extend DeviceContext = strcat("Model:", DeviceModel_s, "|Ver:", CurrentFirmwareVersion_s, "|Channel:", UpdateChannel_s)
| project TimeGenerated, UAVId_s, DetectionType, ViolationCount, DeviceContext, FirmwareHash_s, SBOM_ComponentList_s
```

### ③ KQL 설명
`leftanti`를 쓰면 등록된 GCS가 탈취당하는 핵심 공격을 통째로 놓침 → `leftouter`로 전환. **권한기반 3체크(미등록/잘못된망/대량재지정)는 슈퍼어드민한테 다 무력화되므로**, 권한과 무관한 4번째 체크(평소 접속 IP·시간대와 다른가)를 추가.
```plain text
// GCS 횡적확산 탐지: 등록된 GCS도 믿지 말고 계속 검사 + 권한기준이 무력화되는 슈퍼어드민은 행동패턴으로 별도 체크

let AuthorizedGCS = _GetWatchlist('Authorized_GCS_Asset_List');  // 인가된 GCS + 각자의 담당 네트워크 목록
let lookback = 30m;
let baselineDays = 30d;

// 계정별 행동 베이스라인 (권한과 무관하게 "평소 어디서/언제 로그인했는지"만 봄)
let LoginBaseline =
    GCS_Network_Logs_CL
    | where TimeGenerated between (ago(baselineDays) .. ago(lookback))
    | summarize KnownIPs = make_set(SourceIp_s), KnownHours = make_set(hourofday(TimeGenerated))
        by Operator_Account_s;

GCS_Network_Logs_CL
| where TimeGenerated > ago(lookback)
| summarize ActualRetaskCount = sum(Retask_Count_d), arg_max(TimeGenerated, Target_Net_s, Operator_Account_s, SourceIp_s)
    by GCS_ID_s
    // 30분 윈도우로 직접 합산 — 원본이 윈도우집계값인지 불명확했던 문제 해결

| join kind=leftouter (AuthorizedGCS) on GCS_ID_s
    // ★ leftouter (leftanti 아님!) — 등록된 GCS도 계속 검사대상으로 살려둠
| extend isUnregisteredGCS = isnull(Network_Segment_s)
| extend isWrongSegment = isnotnull(Network_Segment_s) and Target_Net_s != Network_Segment_s
| extend isMassRetask = ActualRetaskCount > 5   // ⚠️ 가상 임계치, P95 기반 보정 필요

| join kind=leftouter (LoginBaseline) on Operator_Account_s
| extend isNewIP = isnotempty(KnownIPs) and not(SourceIp_s in (KnownIPs))
| extend isUnusualHour = isnotempty(KnownHours) and not(hourofday(TimeGenerated) in (KnownHours))
| extend isAnomalousLogin = isNewIP or isUnusualHour
    // ★ 추가: 권한(담당망/슈퍼어드민 여부)과 무관하게 "평소 패턴과 다른가"만 봄
    //   슈퍼어드민은 위 세 체크에 다 안 걸려도, 이건 걸릴 수 있음

| extend ViolationCount = toint(isUnregisteredGCS) + toint(isWrongSegment) + toint(isMassRetask) + toint(isAnomalousLogin)
| where ViolationCount > 0
| extend Reason = case(
    ViolationCount > 1, "Multiple_Violations",
    isUnregisteredGCS, "Unregistered_GCS",
    isWrongSegment, "Wrong_Network_Segment",
    isMassRetask, "Mass_Retask_Spike",
    "Anomalous_Login_Pattern"
)
| project TimeGenerated, GCS_ID_s, Operator_Account_s, Target_Net_s, Network_Segment_s, ActualRetaskCount, SourceIp_s, Reason
```

```plain text
// UGV 노획 탐지: "통신은 끊겼는데 물리적으로 움직인다" + AI 의도판단을 결합해서 등급 나누기
// ★ Intent_Type → Intent_Type_s 컬럼명 수정, Signal_Source_s 명시적 체크 추가

let Thresholds = _GetWatchlist('UGV_Threshold_List');  // 기체별 평시 진동기준 + zeroize용 confidence 컷오프
let lookback = 5m;

let PhysicalAnomaly =
    UGV_Telemetry_Logs_CL
    | where TimeGenerated > ago(lookback)
    | join kind=inner (Thresholds) on UGV_ID_s
        // 기준값 가져오기 (여기는 inner — 기준값 없는 기체는 판단 자체가 불가능하므로 제외)
    | extend TotalMovement = sqrt(pow(Accel_X_d, 2) + pow(Accel_Y_d, 2) + pow(Accel_Z_d, 2))
        // 3축 가속도를 하나의 "흔들림 크기" 숫자로 합치기 (피타고라스 정리처럼)
    | where Heartbeat_Status_s == "Disconnected"
        and Signal_Source_s == "Beacon"
        and TotalMovement > Noise_Threshold_d
        // ★ Signal_Source_s 명시 추가: C2가 끊겼는데 텔레메트리가 들어오는 건 "비콘 채널" 가정이었음
        //   가정을 코드로도 강제 — 다른 채널에서 잘못 들어온 행은 제외
    | project TimeGenerated, UGV_ID_s, TotalMovement, Noise_Threshold_d, Confidence_Threshold_d;

let AIIntent =
    AI_Agent_Intent_Logs_CL
    | where TimeGenerated > ago(lookback)
    | where Intent_Type_s == "Physical_Capture_Attempt"
        // ★ Intent_Type → Intent_Type_s (스키마 표기와 불일치하던 실제 버그)
    | summarize arg_max(Confidence_Score_d, Reasoning_s) by UGV_ID_s
    | project-rename LastIntentConfidence = Confidence_Score_d, LastReasoning = Reasoning_s;
        // 기체별로 가장 높은 confidence와 그때의 판단근거를 같이 가져옴 (OSCAL 증거용)

PhysicalAnomaly
| join kind=leftouter (AIIntent) on UGV_ID_s
    // ★ leftouter — AI 판단이 아직 없어도 버리지 않고 살려둠
| extend Severity = case(
    isnull(LastIntentConfidence), "M_PendingAIJudgment",
    LastIntentConfidence >= Confidence_Threshold_d, "H_ZeroizeTrigger",
    "L_Monitor"
)
| project TimeGenerated, UGV_ID_s, TotalMovement, LastIntentConfidence, LastReasoning, Severity,
    Action = case(
        Severity == "H_ZeroizeTrigger", "Trigger_PB-UGV-CONTAIN-07_Zeroize",
        Severity == "M_PendingAIJudgment", "Escalate_to_Investigation_Agent",
        "Log_Only"
    )
```

## S8. 온보드 표적인식 AI 적대적 공격
### ① 시나리오 설명

### ③ KQL 설명
"센서 간 즉시 불일치"와 "신뢰도 분포의 통계적 이상"을 독립된 두 경로로 union (S2와 같은 패턴). 역광 등 알려진 어려운 환경은 Watchlist 허용범위 내에서 예외 처리.
```plain text
// 온보드 AI 적대적 공격 탐지: EO/IR 두 센서가 같은 표적을 다르게 보거나, 신뢰도 패턴이 이상하면 의심

let EnvBaseline = _GetWatchlist('EOIR_Model_Baseline_List');  // 환경조건별 "이정도 센서간 격차는 정상" 허용폭
let baselineWindow = 24h;
let alertWindow = 10m;

// [경로 A] 즉시 탐지: 센서 간 표적 불일치
let CrossSensorMismatch =
    PayloadEOIR_Inference_CL
    | where TimeGenerated > ago(alertWindow)
    | extend isClassMismatch = (EO_TargetClass_s != IR_TargetClass_s)
        // EO(전자광학)와 IR(적외선)이 같은 물체를 다른 종류로 분류했나
    | extend ConfidenceGap = abs(EO_Confidence_d - IR_Confidence_d)
        // 두 센서의 확신도 차이
    | join kind=leftouter (EnvBaseline) on EnvCondition_s
        // 지금 환경(역광·저조도 등)에서 원래 허용되는 격차가 뭔지 대조
    | extend isEnvException = isnotnull(MaxConfidenceGap_d) and ConfidenceGap <= MaxConfidenceGap_d
        // 알려진 환경조건 + 그 허용범위 안 = "환경 때문"
    | where isClassMismatch and not(isEnvException)
        // 분류가 다르고, 환경 탓도 아니면 통과
    | extend DetectionType = "Cross_Sensor_Mismatch",
             Severity = iff(WeaponGuidanceLinked_b == true, "H_Escalate", "M_Monitor")
        // 무장유도와 연결된 표적이면 즉시 상향
    | project TimeGenerated, PlatformId_s, DetectionType, Severity,
        Detail = strcat("EO:", EO_TargetClass_s, "|IR:", IR_TargetClass_s, "|Gap:", ConfidenceGap);

// [경로 B] 통계 탐지: 신뢰도 분포 이상 (베이스라인 24h로 학습)
let ConfidenceAnomaly =
    PayloadEOIR_Inference_CL
    | where TimeGenerated > ago(baselineWindow)
    | make-series AvgEOConf = avg(EO_Confidence_d) on TimeGenerated step 5m by PlatformId_s
        // 기체별 EO 신뢰도 흐름 만들기
    | extend (ConfAnomalies, Score, Baseline) = series_decompose_anomalies(AvgEOConf, 3.0)
        // 평소와 다르게 신뢰도가 튀는 지점 찾기 (적대적 패치가 신뢰도를 비정상적으로 흔들 수 있음)
    | mv-expand TimeGenerated, AvgEOConf, ConfAnomalies
    | where TimeGenerated > ago(alertWindow) and ConfAnomalies == 1
        // 최근 10분 + 이상치인 것만
    | extend DetectionType = "Confidence_Distribution_Anomaly", Severity = "M_Monitor"
    | project TimeGenerated, PlatformId_s, DetectionType, Severity, Detail = strcat("AvgConf:", AvgEOConf, "|Score:", Score);

union CrossSensorMismatch, ConfidenceAnomaly
    // 두 경로 중 어디서든 잡히면 알람
| sort by TimeGenerated desc
```

### ③ KQL 설명
전제: S1\~S11 각 탐지 룰이 공통으로 결과를 적재하는 통합 스트림(`SOC_Alert_Stream_CL`, 컬럼: `AssetId_s`, `ScenarioId_s`)이 있다고 가정. (이 통합 스트림 자체가 아직 설계 안 됐으면 별도 논의 필요)
```plain text
// 군집 포화 탐지: 짧은 시간에 여러 자산에서 알람이 동시다발로 쏟아지면 = SOC 마비시키려는 공격

let CapacityBaseline = _GetWatchlist('SOC_Capacity_Baseline_List');  // 시간대별 "평소 이정도 알람은 정상" 기준
let burstWindow = 5m;
let minDistinctAssets = 5;   // 최소 몇 개 자산이 같이 떠야 "군집"으로 볼지

SOC_Alert_Stream_CL   // S1~S11 모든 룰의 알람이 모이는 통합 스트림 (가정)
| where TimeGenerated > ago(burstWindow)
| extend CurrentShift = iff(hourofday(TimeGenerated) between (8 .. 18), "BusinessHours", "NightShift")
    // 지금이 업무시간인지 야간인지 구분 (시간대별로 평시 알람량이 다르니까)
| summarize AlertCount = count(), DistinctAssets = dcount(AssetId_s), Scenarios = make_set(ScenarioId_s) by CurrentShift
    // 최근 5분간 총 알람수, 서로 다른 자산 수, 어떤 시나리오들이 떴는지 집계
| join kind=leftouter (CapacityBaseline) on $left.CurrentShift == $right.TimeWindow_s
    // 이 시간대의 "평소 알람량" 기준 가져오기
| extend isExerciseException = (Condition_s == "Scheduled_Exercise")
    // 사전 등록된 훈련이면 예외
| extend isVolumeSpike = AlertCount > (3 * MaxNormalAlertRate_d)
    // 평시 대비 3배 넘게 알람이 몰렸나
| extend isMultiAxis = DistinctAssets >= minDistinctAssets
    // 한 자산만 시끄러운 게 아니라 여러 자산이 같이 튀었나
| where isVolumeSpike and isMultiAxis and not(isExerciseException)
    // 폭증 + 다축 + 훈련 아님, 셋 다 맞아야 통과
| extend TriggerReason = "Swarm_Saturation_AlertStorm"
| project TimeGenerated = now(), CurrentShift, AlertCount, DistinctAssets, Scenarios, TriggerReason
```

### ③ KQL 설명
"기상으로 인한 동시 장애"와 "관리채널을 통한 악성 업데이트로 인한 동시 장애"를 `MgmtChannelAnomaly_b`와 업데이트 소스로 구분. 기본값(0.1)을 숨기지 않고 명시.
```plain text
// SATCOM 단말 무력화 탐지: 여러 단말이 동시에 죽으면 "날씨 탓"과 "관리망 공격"을 구분

let MgmtBaseline = _GetWatchlist('SATCOM_Mgmt_Baseline_List');  // 그룹별 "평소 이정도 실패율은 정상" 기준
let burstWindow = 5m;

SATCOM_Mgmt_Logs_CL
| where TimeGenerated > ago(burstWindow)
| summarize TotalTerminals = dcount(TerminalId_s),                                       // 이 그룹의 전체 단말 수
            FailedTerminals = dcountif(TerminalId_s, ConnectionStatus_s == "Failed"),     // 그 중 접속실패한 단말 수
            AnyMgmtAnomaly = max(MgmtChannelAnomaly_b),                                   // 관리채널 이상이 하나라도 있었나
            UnofficialUpdateSeen = max(FirmwareUpdateSource_s != "Official")              // 비공식 경로 업데이트가 있었나
    by TerminalGroup_s
| extend FailureRate = todouble(FailedTerminals) / todouble(TotalTerminals)
    // 이 그룹에서 몇 %가 실패했는지
| join kind=leftouter (MgmtBaseline) on TerminalGroup_s
    // 이 그룹의 평소 정상 실패율이 얼마인지 대조
| extend isWeatherException = isnotnull(MaxNormalFailureRate_d) and FailureRate <= MaxNormalFailureRate_d and not(AnyMgmtAnomaly)
    // 평소 범위 안이고 + 관리채널도 정상이면 = "그냥 날씨 탓"
// ⚠️ Watchlist 미매칭 시 기본 10%로 가정 — 가상치, 실데이터 확보 후 보정 필요
| extend isMassFailure = FailureRate > coalesce(MaxNormalFailureRate_d, 0.10)
    // 기준치(없으면 기본 10%) 넘게 실패했나
| where isMassFailure and not(isWeatherException)
    // 날씨 탓이 아니면서 대량실패면 통과
| extend TriggerReason = case(
    AnyMgmtAnomaly and UnofficialUpdateSeen, "Malicious_Mgmt_Update",   // 관리채널 이상 + 비공식 업데이트 = 가장 의심
    AnyMgmtAnomaly, "Mgmt_Channel_Anomaly",                              // 관리채널 이상만
    "Unexplained_Mass_Failure"                                           // 원인 불명의 대량실패
)
| project TimeGenerated = now(), TerminalGroup_s, TotalTerminals, FailedTerminals, FailureRate, TriggerReason
```

### ③ KQL 설명
S6에서 배운 `leftanti` 함정을 또 안 반복하도록 `leftouter` + 우측 전용 컬럼 null 체크로 설계. join 시 양쪽 테이블에 같은 컬럼명(`AppSignatureHash_s`)이 있어서 Kusto가 자동으로 `_1` 접미사를 붙이는 점 주의.
```plain text
// 모바일 GCS 침해 탐지: 비인가 앱 사용과, 한 토큰이 여러 기기에서 동시에 쓰이는 것을 따로 잡기

let AppWhitelist = _GetWatchlist('Mobile_GCS_App_Whitelist');  // 인가된 앱 패키지 + 서명 해시 목록
let lookback = 15m;

// [경로 A] 비인가 앱 / 서명 불일치
let UnauthorizedApp =
    MobileGCS_Activity_CL
    | where TimeGenerated > ago(lookback)
    | join kind=leftouter (AppWhitelist) on AppPackageId_s
        // ★ leftouter (S6의 leftanti 실수를 여기선 안 반복) — 등록된 앱도 계속 검사
    // ※ 양쪽에 AppSignatureHash_s가 있어 우측은 AppSignatureHash_s1 로 자동 변경됨
    | extend isUnregisteredApp = isnull(AppSignatureHash_s1)
        // 화이트리스트 전용 컬럼으로 매칭 여부 체크 (join key 아님)
    | extend isSignatureMismatch = isnotnull(AppSignatureHash_s1) and AppSignatureHash_s != AppSignatureHash_s1
        // 앱은 등록돼 있는데 서명 해시가 다르다 = 위변조 의심
    | where isUnregisteredApp or isSignatureMismatch
    | extend DetectionType = iff(isUnregisteredApp, "Unauthorized_App", "App_Signature_Mismatch")
    | project TimeGenerated, DeviceId_s, AppPackageId_s, DetectionType;

// [경로 B] 세션 토큰 재사용 (동일 토큰이 복수 기기에서 동시 관측)
let TokenReuse =
    MobileGCS_Activity_CL
    | where TimeGenerated > ago(lookback)
    | summarize DeviceCount = dcount(DeviceId_s), Devices = make_set(DeviceId_s) by SessionToken_s
        // 같은 토큰을 쓴 기기가 몇 개인지 세기
    | where DeviceCount > 1
        // 토큰 하나가 2개 이상 기기에서 쓰였다 = 같은 시간에 두 곳에 있을 수 없으니 탈취 확정 (통계적 추정 아니라 논리적 사실)
    | extend DetectionType = "Session_Token_Reuse"
    | project TimeGenerated = now(), SessionToken_s, DeviceCount, Devices, DetectionType;

union UnauthorizedApp, TokenReuse
    // 두 경로 중 어디서든 잡히면 알람
| sort by TimeGenerated desc
```
